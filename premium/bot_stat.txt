    // Counter animation
    function animateCounter(elementId, target, duration = 2500, delay = 0, formatter = (n) => n.toLocaleString()) {
      setTimeout(() => {
        const element = document.getElementById(elementId);
        const start = performance.now();
        
        function update(currentTime) {
          const elapsed = currentTime - start;
          const progress = Math.min(elapsed / duration, 1);
          const easeOut = 1 - Math.pow(1 - progress, 4);
          const current = Math.floor(easeOut * target);
          
          element.textContent = formatter(current);
          
          if (progress < 1) {
            requestAnimationFrame(update);
          } else {
            element.textContent = formatter(target);
          }
        }
        
        requestAnimationFrame(update);
      }, delay);
    }

    function formatK(num) {
      if (num >= 1000) {
        return (num / 1000).toFixed(num >= 10000 ? 0 : 1) + 'K+';
      }
      return num.toLocaleString() + '+';
    }

    // Start counters
    animateCounter('main-counter', 11202248, 3000, 300);
    animateCounter('servers-counter', 6500, 2500, 600, formatK);
    animateCounter('users-counter', 100000, 2500, 800, formatK);
    animateCounter('premium-counter', 2000, 2500, 1000, formatK);

    const container = document.getElementById('canvas-container');
    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(50, window.innerWidth / window.innerHeight, 0.1, 1000);
    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
    
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    container.appendChild(renderer.domElement);

    camera.position.z = 8;

    // Continent data (simplified lat/lng ranges)
    const continents = [
      { latMin: 25, latMax: 70, lngMin: -170, lngMax: -50 },   // North America
      { latMin: -55, latMax: 12, lngMin: -80, lngMax: -35 },   // South America
      { latMin: 35, latMax: 70, lngMin: -10, lngMax: 40 },     // Europe
      { latMin: -35, latMax: 37, lngMin: -20, lngMax: 50 },    // Africa
      { latMin: 5, latMax: 75, lngMin: 40, lngMax: 145 },      // Asia
      { latMin: -45, latMax: -10, lngMin: 110, lngMax: 155 },  // Australia
    ];

    function isLand(lat, lng) {
      for (const c of continents) {
        if (lat >= c.latMin && lat <= c.latMax && lng >= c.lngMin && lng <= c.lngMax) {
          return true;
        }
      }
      return false;
    }

    // Create hexagonal dots for globe
    const globeGroup = new THREE.Group();
    const radius = 3;
    const dotCount = 10000;

    for (let i = 0; i < dotCount; i++) {
      const phi = Math.acos(-1 + (2 * i) / dotCount);
      const theta = Math.sqrt(dotCount * Math.PI) * phi;
      
      const lat = 90 - (phi * 180) / Math.PI;
      const lng = ((theta * 180) / Math.PI) % 360 - 180;
      
      const x = radius * Math.cos(theta) * Math.sin(phi);
      const y = radius * Math.sin(theta) * Math.sin(phi);
      const z = radius * Math.cos(phi);
      
      const land = isLand(lat, lng);
      
      const geometry = new THREE.CircleGeometry(land ? 0.035 : 0.02, 6);
      const material = new THREE.MeshBasicMaterial({
        color: land ? 0xa78bfa : 0x3b3b6d,
        transparent: true,
        opacity: land ? 0.9 : 0.25,
        side: THREE.DoubleSide
      });
      
      const dot = new THREE.Mesh(geometry, material);
      dot.position.set(x, y, z);
      dot.lookAt(0, 0, 0);
      globeGroup.add(dot);
    }

    scene.add(globeGroup);

    // Atmosphere glow spheres
    const atmosphereGeometry = new THREE.SphereGeometry(3.1, 64, 64);
    const atmosphereMaterial = new THREE.MeshBasicMaterial({

    const atmosphere = new THREE.Mesh(atmosphereGeometry, atmosphereMaterial);

    for (let i = 0; i < 80; i++) {

      const z1 = (Math.random() - 0.5) * 10 - 8;

      const y2 = y1 + (Math.random() - 0.5) * 4;

    const dotGeometry = new THREE.BufferGeometry();
    const dotPositions = [];
    for (let i = 0; i < 150; i++) {
      dotPositions.push(
        (Math.random() - 0.5) * 20,
        (Math.random() - 0.5) * 20,
        (Math.random() - 0.5) * 10 - 8
      );
    }
    dotGeometry.setAttribute('position', new THREE.Float32BufferAttribute(dotPositions, 3));
    const dotMaterial = new THREE.PointsMaterial({ color: 0x60a5fa, size: 0.08, transparent: true, opacity: 0.5 });
    const networkDots = new THREE.Points(dotGeometry, dotMaterial);
    scene.add(networkDots);

    // Animation
    let isDragging = false;
    let previousMousePosition = { x: 0, y: 0 };

    document.addEventListener('mousedown', () => isDragging = true);
    document.addEventListener('mouseup', () => isDragging = false);
    document.addEventListener('mousemove', (e) => {
      if (isDragging) {
        const deltaX = e.clientX - previousMousePosition.x;
        const deltaY = e.clientY - previousMousePosition.y;
        globeGroup.rotation.y += deltaX * 0.005;
        globeGroup.rotation.x += deltaY * 0.005;
      }
      previousMousePosition = { x: e.clientX, y: e.clientY };
    });

    function animate() {
      requestAnimationFrame(animate);
      
      if (!isDragging) {
        globeGroup.rotation.y += 0.001;
      }
      
      atmosphere.scale.setScalar(1 + Math.sin(Date.now() * 0.0003) * 0.02);
      networkDots.rotation.y = Math.sin(Date.now() * 0.0001) * 0.1;
      
      renderer.render(scene, camera);
    }
    animate();

    // Handle resize
    window.addEventListener('resize', () => {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    });
  </script>
